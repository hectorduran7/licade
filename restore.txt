    function sanitizeHTML(str) {
        if (!str) return '';
        return str.replace(/[&<>]/g, function(m) {
            if (m === '&') return '&amp;';
            if (m === '<') return '&lt;';
            if (m === '>') return '&gt;';
            return m;
        });
    }
    
    function sanitizeText(str) {
        if (!str) return '';
        return str.replace(/[<>]/g, function(m) {
            if (m === '<') return '&lt;';
            if (m === '>') return '&gt;';
            return m;
        });
    }
    
    function sanitizeSubjectName(name) {
        if (!name) return '';
        let cleaned = name.replace(/[^\w\s\u00C0-\u00FF\-\.\,\(\)\:]/gi, '');
        if (cleaned.length > 100) cleaned = cleaned.substring(0, 100);
        return cleaned.trim();
    }

        const isNumeric = !isNaN(gradeNum) && gradeNum >= 1 && gradeNum <= 10;

            return;
        }

        if (!isNumeric && !isApproved) { 
            alert("Ingresá una nota válida del 1 al 10 o la letra 'A'."); 
            return; 
        }
        
        subj = sanitizeSubjectName(subj);
        if (!subj) {
            alert('El nombre de la materia contiene caracteres no válidos.');
            return;
        }

        entries.push({

            } else {

            }
        }
    };

    function saveAndRender() {
        const list = document.getElementById('gradeList');
        if (entries.length === 0) {
            list.innerHTML = '<div style="text-align: center; padding: 40px 20px; color: #9ca3af;">' +
                '<i data-lucide="inbox" class="icon-lg" style="width: 40px; height: 40px; margin-bottom: 10px; display: block;"></i>' +
                '<p>No hay materias cargadas aún. Agregá tu primera materia arriba.</p>' +
            '</div>';
        } else {
            list.innerHTML = entries.map(function(entry, index) {
                return '<div class="grade-item">' +
                    '<span style="font-weight:500;">' + escapeHtml(entry.subj) + '</span>' +
                    '<span style="display:flex; align-items:center; gap:10px;">' +
                        '<b style="color:var(--primary);">' + escapeHtml(String(entry.grade)) + '</b>' +
                        '<span class="btn-del" onclick="removeEntry(' + index + ')" title="Eliminar materia">' +
                            '<i data-lucide="trash-2" class="icon-sm"></i>' +
                        '</span>' +
                    '</span>' +
                '</div>';
            }).join('');
        }

        // Filtro 1, 2 y 3: Promedio exacto agrupando por materia y filtrando textos/ausentes
        var notasPorMateria = {};
        entries.forEach(function(entry) {
            var gradeStr = String(entry.grade).trim().toLowerCase();
            // Ignorar textos, vacíos y ausentes
            if (gradeStr === 'a' || gradeStr === 'aprobado' || gradeStr === 'ausente' || gradeStr === '') return;
            
            var gradeNum = Number(entry.grade);
            if (!isNaN(gradeNum)) {
                var subj = entry.subj.trim().toLowerCase();
                // Si la materia ya tiene una nota cargada, guardamos la más alta
                if (!notasPorMateria[subj] || gradeNum > notasPorMateria[subj]) {
                    notasPorMateria[subj] = gradeNum;
                }
            }
        });

        var sum = 0;
        var count = 0;
        for (var key in notasPorMateria) {
            sum += notasPorMateria[key];
            count++;
        }
        var average = count > 0 ? (sum / count).toFixed(2) : '0.00';

        document.getElementById('avgOut').innerText = average;
        
        var completed = entries.length;
        var percentage = Math.min((completed / totalMaterias) * 100, 100).toFixed(1);
        document.getElementById('progressBar').style.width = percentage + '%';
